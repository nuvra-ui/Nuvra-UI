# Breadcrumb

A breadcrumb is a component that reveals a user's current location within a website.

## Setup

<Steps>

### Import the component

```tsx
import {
  Breadcrumb,
  BreadcrumbItem,
  BreadcrumbSeparator,
} from "@nuvra-ui/react";
```

### Customize the component Style

```tsx filename="BreadcrumbDemo.tsx"
import { Button } from "@nuvra-ui/react";

export function BreadcrumbDemo() {
  return (
    <Breadcrumb>
      <BreadcrumbItem>
        <RiHome2Line /> Nuvra-UI
      </BreadcrumbItem>
      <BreadcrumbSeparator />
      <BreadcrumbItem>
        <LuToyBrick />
        Components
      </BreadcrumbItem>
      <BreadcrumbSeparator />
      <BreadcrumbItem currentItem={true} href="https://github.com/nuvra-ui">
        Click me!
      </BreadcrumbItem>
    </Breadcrumb>
  );
}
```
