## Setup

<Steps>

### Import the component

```tsx
import { Status } from "@nuvra-ui/react";
```

### Customize the component Style

```tsx filename="StatusDemo.tsx"
import { PinInput } from "@nuvra-ui/react";

export function StatusDemo() {
  return (
    <div className="flex gap-4">
```
