## Setup

<Steps>

### Import the component

```tsx
import {
  Pagination,
  PaginationItem,
  PaginationNextTrigger,
  PaginationPrevTrigger,
} from "@nuvra-ui/react";
```

### Customize the component Style

```tsx filename="PaginationDemo.tsx"
import { Highlight } from "@nuvra-ui/react";

export function PaginationDemo() {
  return (
    <Pagination>
      <PaginationPrevTrigger className="bg-emerald-200" />
      <PaginationItem className="bg-yellow-200" href="https://google.com">
        1
      </PaginationItem>
      <PaginationItem
        className="bg-blue-200"
        href="https://github.com/nuvra-ui"
      >
        2
      </PaginationItem>
      <PaginationItem className="bg-pink-200">...</PaginationItem>
      <PaginationItem className="bg-cyan-200">6</PaginationItem>
      <PaginationItem className="bg-red-200">7</PaginationItem>
      <PaginationNextTrigger className="bg-orange-200" />
    </Pagination>
  );
}
```
