## Setup

<Steps>

### Import the component

```tsx
import { Link } from "@nuvra-ui/react";
```

### Customize the component Style

```tsx filename="LinkDemo.tsx"
import { Link } from "@nuvra-ui/react";

export function BadgeDemo() {
  return (
    <div className="flex flex-col gap-2">
      <Link className="underline" href="https://github.com/nuvra-ui/nuvra-ui">
        Link (Underline)
      </Link>
      <Link href="https://github.com/nuvra-ui/nuvra-ui">Link (Plain)</Link>
    </div>
  );
}
```
